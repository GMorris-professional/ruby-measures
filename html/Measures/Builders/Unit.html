def self.build(system_of_units)
  builder = new(system_of_units)
  yield(builder)
  Unit.new(aliases: builder.get_aliases, quantity: builder.get_quantity, system: builder.system,
           symbol: builder.get_symbol, prefix: builder.get_prefix)
end
